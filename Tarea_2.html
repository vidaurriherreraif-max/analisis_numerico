#  Función 
def punto_fijo(g, p0, tol=1e-5, itmax=500):
    for i in range(itmax):
        p_new = g(p0)
        if np.abs(p_new - p0) < tol:
            return p_new, i + 1
        p0 = p_new
    raise RuntimeError(f"No se encontró un punto fijo en {itmax} iteraciones.")

def punto_fijo(g, p0, tol, itmax):
    tabla = [(0, p0, None, None)]
    for i in range(1, itmax+1):
        p = g(p0)
        err_abs = abs(p - p0)
        err_rel = err_abs / abs(p) if p != 0 else None
        tabla.append((i, p, err_abs, err_rel))
        if err_abs < tol:
            break
        p0 = p
    return tabla

    tabla = [(0, p0, None, None)]

        p = g(p0)
        err_abs = abs(p - p0)
        err_rel = err_abs / abs(p) if p != 0 else None
        tabla.append((i, p, err_abs, err_rel))
        if err_abs < tol:
            break
        p0 = p
    return tabla

#  Coeficientes de Taylor: f^(k)(0) / k!
# Derivadas de e^x*cos(x) evaluadas en x=0:
# f(x)   = e^x cos(x)         → f(0)   =  1
# f'(x)  = e^x(cos-sin)       → f'(0)  =  1
# f''(x) = -2e^x sin(x)       → f''(0) =  0
# f'''(x)= -2e^x(sin+cos)     → f'''(0)= -2  ... pero usamos diff numérica general

def taylor_coef(f, x0, n, h=1e-5):
    """Coeficientes a_k = f^(k)(x0) / k! usando diferencias finitas."""
    coefs = []
    for k in range(n + 1):
        # Derivada k-ésima por diferencias finitas centradas
        dk = 0
        for j in range(k + 1):
            dk += ((-1)**(k - j)) * math.comb(k, j) * f(x0 + j * h)
        dk /= h**k
        coefs.append(dk / math.factorial(k))
    return coefs

#  Polinomio de Taylor 
def taylor_poly(coefs, x, x0):
    resultado = 0
    for k, ak in enumerate(coefs):
        resultado += ak * (x - x0)**k
    return resultado

#  Función CRRA 
f = lambda x: x**(1 - gamma) / (1 - gamma)

#  Derivadas de f(x) = x^(1-γ) / (1-γ) 
# f^(k)(x) = (1-γ)(1-γ-1)···(1-γ-k+1) · x^(1-γ-k) / (1-γ)
#           = [(1-γ-1)(1-γ-2)···(1-γ-k+1)] · x^(1-γ-k)
#
# Coeficiente a_k = f^(k)(x0) / k!
# Con x0=1: x0^(1-γ-k) = 1
# Entonces a_k = prod_{j=1}^{k}(1-γ-j+1) / (1-γ) ... calculamos directo

def derivada_k(k, gamma, x0):
    """f^(k)(x0) para f(x) = x^(1-gamma)/(1-gamma)"""
    if k == 0:
        return x0**(1 - gamma) / (1 - gamma)
    # f^(k)(x) = [prod_{j=0}^{k-1} (1-gamma-j)] * x^(1-gamma-k) / (1-gamma)
    prod = 1.0
    for j in range(k):

    return prod * x0**(1 - gamma - k) / (1 - gamma)

#  Derivadas de f(r) = (1 + r/m)^m - 1 
# f^(k)(r) = m(m-1)···(m-k+1) · (1/m)^k · (1 + r/m)^(m-k)
# En r0=0: (1 + 0)^(m-k) = 1
# f^(k)(0) = m! / (m-k)! · (1/m)^k   para k <= m
#           = prod_{j=0}^{k-1}(m-j) / m^k

def derivada_k(k, m, r0):
    if k == 0:
        return f(r0)
    prod = 1.0
    for j in range(k):
        prod *= (m - j)
    return prod * (1/m)**k * (1 + r0/m)**(m - k)

coefs = []
print(f"Coeficientes de Taylor a_k = f^(k)(r0)/k!  [m={m}, r0={r0}]:")
print(f"{'k':>3}  {'f^(k)(0)':>14}  {'a_k':>14}")
print("-" * 36)
for k in range(n + 1):
    dk = derivada_k(k, m, r0)
    ak = dk / math.factorial(k)
    coefs.append(ak)
    print(f"{k:>3}  {dk:>14.8f}  {ak:>14.8f}")

# Polinomio de Taylor 
def taylor_poly(coefs, r, r0):
    return sum(ak * (r - r0)**k for k, ak in enumerate(coefs))

print(f"\nT_{n}(r) centrado en r0={r0}:")
terminos = []
for k, ak in enumerate(coefs):
    if abs(ak) > 1e-12:
        if k == 0:
            terminos.append(f"{ak:.6f}")
        elif k == 1:
            terminos.append(f"{ak:+.6f}r")
        else:
            terminos.append(f"{ak:+.6f}r^{k}")
print("T5(r) = " + " ".join(terminos))

#  Gráfica 
r_vals = np.linspace(0, 0.5, 500)
T5     = np.array([taylor_poly(coefs, r, r0) for r in r_vals])

plt.figure(figsize=(7, 5))
plt.plot(r_vals, f(r_vals),  color='darkred',   linewidth=2,
         label=r'$f(r) = \left(1+\frac{r}{12}\right)^{12}-1$')
plt.plot(r_vals, T5,          color='steelblue', linewidth=2,
         linestyle='--', label=r'$T_5(r)$')
plt.plot(r0, f(r0), marker='o', ms=8, color='darkorange',
         label=f'$r_0 = {r0}$')
plt.xlabel('Tasa nominal r')
plt.ylabel('Tasa efectiva anual')
plt.title(r'Ejercicio 9: Taylor Tasa de Interés, $m=12$, $r_0=0$, $n=5$')
plt.legend()
